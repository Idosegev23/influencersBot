# P2 вАФ Voice + Models: Hebrew STT with confidence + config-driven dual-provider model layer

## Goal
Replace the agent WhatsApp voice path's misuse of the document quote-parser (`parseAudioWithGemini({documentType:'quote'})`) with a **dedicated Hebrew STT** returning `{text, confidence, provider}`, add a **confidence-gated read-back** before any money action, and replace the single `gpt-5-nano` `chat()` call in the agent money path with a **config-driven dual-provider (OpenAI + Gemini) structured-output layer** that mutually falls back. Every model ID is config-driven and verified against the live SDK before it ships.

## Architecture
- `src/lib/stt/` вАФ voice: `transcribeHebrew()` (I/O, provider-injectable) + `confidence.ts` (pure, DB-free).
- `src/lib/llm/` вАФ `config.ts` (model matrix, env-overridable) + `index.ts` (`extractStructured` / `routeIntent`, provider-injectable, mutual fallback, strict-JSON on OpenAI вЖФ `responseSchema` on Gemini).
- `src/lib/crm/wa-conversation.ts` вАФ new `awaiting_voice_confirm` stage + pure `planVoiceIntake()`; money-lane extraction re-wired to `extractStructured`.
- `src/app/api/webhooks/whatsapp/route.ts` вАФ voice path swaps to `transcribeHebrew`, stores `stt_confidence`/`stt_provider`.
- `scripts/verify-agent-model-ids.mjs` вАФ pre-deploy gate: `models.list()` on both providers.

Pure helpers (`confidence.ts`) are separated from I/O (`transcribeHebrew.ts`) and DB (`wa-conversation.ts`) per repo convention; unit tests target the pure/injectable seams and never hit the network.

## Tech Stack
- OpenAI SDK v6.27 (`responses.create` structured outputs; `audio.transcriptions.create`; `models.list`), verified default IDs `gpt-5` / `gpt-5-nano`.
- Google `@google/genai` v1.38 (`models.generateContent` native-audio + `responseSchema`; `models.list`), verified default IDs `gemini-3.1-pro-preview` (MODELS.COMPLEX) / `gemini-3.5-flash` (MODELS.CHAT_FAST).
- Vitest (`tests/unit/*.test.ts`, `@/` alias). Supabase MCP `apply_migration` for the two new columns. TypeScript `strict:false` вЖТ run `npm run type-check` separately.

## Global Constraints
- **No WhatsApp buttons / list-pickers.** The read-back is a text echo; the confirm reply is free-form, interpreted in context (reuse P1's model-in-context confirm; STT read-back adds the trigger, not a new UI).
- **No hardcoded model ID may 404.** All IDs come from `laneConfig()` (env вЖТ repo-verified default). `scripts/verify-agent-model-ids.mjs` must pass before deploy.
- Money normalization is single-sourced (`normalizeAmount`, from P0). Idempotency/`{reply,outcome}`/`crm_agent_wa_log` are from P0; async worker + per-agent lock + free-form read-back stage are from P1.

```ts
/**
 * Config-driven model matrix for the agent WhatsApp lanes (P2).
 * Every id is env-overridable; the DEFAULTS are ids already verified live in this
 * repo (google-client MODELS / openai.ts). Aspirational ids (gpt-5.5, gpt-4o-transcribe)
 * are wired via env and gated by scripts/verify-agent-model-ids.mjs before deploy.
 */
import { MODELS } from '@/lib/ai/google-client';

export type Lane = 'money' | 'router' | 'stt';
export interface ModelChoice { provider: 'openai' | 'gemini'; model: string; }
export interface LaneConfig { primary: ModelChoice; fallback: ModelChoice; }

const env = (k: string, d: string) => (process.env[k] || d).trim();

/** Repo-verified fallbacks: gpt-5 / gpt-5-nano (openai.ts), MODELS.COMPLEX / CHAT_FAST (google-client). */
export const AGENT_MODELS: Record<Lane, LaneConfig> = {
  // Money lane: highest stakes вЖТ strong reasoning + strict JSON. Spec target GPT-5.5;
  // default falls back to the repo-verified gpt-5 until verify-script confirms 5.5.
  money: {
    primary:  { provider: 'openai', model: env('AGENT_MODEL_MONEY', 'gpt-5') },
    fallback: { provider: 'gemini', model: env('AGENT_MODEL_MONEY_FALLBACK', MODELS.COMPLEX) },
  },
  // Router: fast+cheap classify. Gemini 3 Flash primary, OpenAI nano fallback.
  router: {
    primary:  { provider: 'gemini', model: env('AGENT_MODEL_ROUTER', MODELS.CHAT_FAST) },
    fallback: { provider: 'openai', model: env('AGENT_MODEL_ROUTER_FALLBACK', 'gpt-5-nano') },
  },
  // STT: Gemini native audio primary, OpenAI transcribe fallback.
  stt: {
    primary:  { provider: 'gemini', model: env('AGENT_MODEL_STT', MODELS.COMPLEX) },
    fallback: { provider: 'openai', model: env('AGENT_MODEL_STT_FALLBACK', 'gpt-4o-transcribe') },
  },
};

export function laneConfig(lane: Lane): LaneConfig {
  return AGENT_MODELS[lane];
}
```

```ts
/**
 * Pure decision layer for confidence-gated voice read-back (P2, spec ¬І3.2/¬І5.4).
 * A money-scale number in a voice note is ALWAYS echoed before recording (echo-
 * before-act), and any low-confidence transcript is echoed regardless of content.
 * DB-free + dependency-light so it is unit-tested without the network.
 */
import { extractNumbers } from '@/lib/crm/wa-interpret';

export interface VoiceReadbackDecision { readback: boolean; echo: string | null; }

const READBACK_THRESHOLD = Number(process.env.AGENT_STT_READBACK_THRESHOLD || '0.75');

/** Money-scale numbers only (>=1000) вАФ qtys like "3 „®„Щ„Ь„°" are not amounts. */
export function criticalAmounts(text: string): number[] {
  return extractNumbers(text || '').filter((n) => n >= 1000);
}

export function decideVoiceReadback(args: { text: string; confidence: number; threshold?: number }): VoiceReadbackDecision {
  const text = (args.text || '').trim();
  const th = args.threshold ?? READBACK_THRESHOLD;
  const amounts = criticalAmounts(text);
  const lowConfidence = args.confidence < th;
  const hasMoney = amounts.length > 0;

  // Money is always echoed; otherwise only echo when we are not confident.
  if (!hasMoney && !lowConfidence) return { readback: false, echo: null };

  const heard = `„©„Ю„Ґ„™„Щ: "${text}"`;
  const money = hasMoney ? ` | „°„Ы„Х„Э: ${amounts.map((n) => n.toLocaleString('en-US')).join(', ')} вВ™` : '';
  return { readback: true, echo: `${heard}${money} вАФ „†„Ы„Х„Я? („Р„§„©„® „Ь„™„І„Я „С„Ю„Щ„Ь„Щ„Э)` };
}
```

**Step 2 вАФ implementation** `src/lib/stt/transcribeHebrew.ts`:
```ts
/**
 * Dedicated Hebrew STT for the agent voice path (P2, spec ¬І3).
 * Primary: Gemini 3.1 Pro native audio (self-reported confidence).
 * Fallback: OpenAI gpt-4o-transcribe (conservative fixed confidence вАФ the endpoint
 * does not expose calibrated token logprobs). Last-resort: Gemini 3.5 Flash.
 * Replaces parseAudioWithGemini({documentType:'quote'}) on the agent voice path only.
 */
import OpenAI from 'openai';
import { getGeminiClient, MODELS } from '@/lib/ai/google-client';
import { laneConfig } from '@/lib/llm/config';

export interface HebrewSttResult { text: string; confidence: number; provider: 'gemini' | 'openai'; model: string; }
export interface SttDeps {
  geminiTranscribe?: (bytes: Uint8Array, mimeType: string, model: string) => Promise<{ text: string; confidence: number }>;
  openaiTranscribe?: (bytes: Uint8Array, mimeType: string, model: string) => Promise<{ text: string; confidence: number }>;
}

const OPENAI_STT_CONFIDENCE = 0.7; // conservative вАФ fallback is itself a mild read-back trigger

async function geminiNativeAudio(bytes: Uint8Array, mimeType: string, model: string): Promise<{ text: string; confidence: number }> {
  const client = getGeminiClient();
  const base64 = Buffer.from(bytes).toString('base64');
  const res = await client.models.generateContent({
    model,
    contents: [{
      role: 'user',
      parts: [
        { inlineData: { data: base64, mimeType } },
        { text: '„™„Ю„Ь„Ь „Р„™ „Ф„Ф„І„Ь„Ш„Ф „С„Ґ„С„®„Щ„™ „Ю„Щ„Ь„Ф „С„Ю„Щ„Ь„Ф. „Ф„Ч„Ц„® JSON „С„Ь„С„У: {"transcription": "<„Ш„І„°„Ш>", "confidence": <0..1>, "language": "he"}. „Р„Ь „™„Х„°„Щ„£ „§„®„©„†„Х„™.' },
      ],
    }],
    config: {
      temperature: 0,
      maxOutputTokens: 4096,
      responseMimeType: 'application/json',
      responseSchema: {
        type: 'object',
        properties: {
          transcription: { type: 'string' },
          confidence: { type: 'number' },
          language: { type: 'string' },
        },
        required: ['transcription', 'confidence'],
      },
    },
  });
  const parsed = JSON.parse((res as any).text || '{}');
  return { text: String(parsed.transcription || '').trim(), confidence: Number(parsed.confidence) || 0 };
}

async function openaiTranscribeImpl(bytes: Uint8Array, mimeType: string, model: string): Promise<{ text: string; confidence: number }> {
  const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
  const ext = (mimeType.split('/')[1] || 'ogg').split(';')[0];
  const file = new File([Buffer.from(bytes)], `voice.${ext}`, { type: mimeType });
  const res: any = await client.audio.transcriptions.create({ file: file as any, model, language: 'he' });
  return { text: String(res?.text || '').trim(), confidence: OPENAI_STT_CONFIDENCE };
}

export async function transcribeHebrew(input: { bytes: Uint8Array; mimeType: string }, deps: SttDeps = {}): Promise<HebrewSttResult> {
  const cfg = laneConfig('stt');
  const gemini = deps.geminiTranscribe || geminiNativeAudio;
  const openai = deps.openaiTranscribe || openaiTranscribeImpl;

  try {
    const r = await gemini(input.bytes, input.mimeType, cfg.primary.model);
    if (r.text) return { text: r.text, confidence: r.confidence, provider: 'gemini', model: cfg.primary.model };
    // empty from Gemini вЖТ try OpenAI before giving up
    const o = await openai(input.bytes, input.mimeType, cfg.fallback.model);
    return { text: o.text, confidence: o.text ? Math.min(o.confidence, OPENAI_STT_CONFIDENCE) : 0, provider: 'openai', model: cfg.fallback.model };
  } catch (e) {
    console.warn('[transcribeHebrew] gemini primary failed, falling back to openai:', (e as any)?.message || e);
    try {
      const o = await openai(input.bytes, input.mimeType, cfg.fallback.model);
      return { text: o.text, confidence: o.text ? Math.min(o.confidence, OPENAI_STT_CONFIDENCE) : 0, provider: 'openai', model: cfg.fallback.model };
    } catch (e2) {
      console.error('[transcribeHebrew] both providers failed:', (e2 as any)?.message || e2);
      return { text: '', confidence: 0, provider: 'openai', model: cfg.fallback.model };
    }
  }
}
```

```ts
 * Dual-provider structured-output layer for the agent WhatsApp lanes (P2, spec ¬І4).
 * OpenAI (responses.create + strict json_schema) and Gemini (generateContent +
 * responseSchema) are each other's fallback. Provider callers are injectable so the
 * fallback/parse logic is unit-tested without the network. Never throws вАФ returns
 * data:null so the caller (money lane) can degrade to a safe path.

import OpenAI from 'openai';
import { getGeminiClient } from '@/lib/ai/google-client';
import { laneConfig, type ModelChoice } from '@/lib/llm/config';

export type JsonSchema = Record<string, any>;
export interface StructuredCall<T> { lane: 'money' | 'router'; system: string; input: string; schema: JsonSchema; schemaName: string; }
export interface StructuredResult<T> { data: T | null; provider: 'openai' | 'gemini'; model: string; usedFallback: boolean; raw: string; }
export interface LlmDeps {
  openaiCall?: (choice: ModelChoice, system: string, input: string, schema: JsonSchema, schemaName: string) => Promise<string>;
  geminiCall?: (choice: ModelChoice, system: string, input: string, schema: JsonSchema, schemaName: string) => Promise<string>;
}

function parseJson<T>(raw: string): T | null {
  try { return JSON.parse(String(raw || '').replace(/```json|```/g, '').trim()) as T; } catch { return null; }
}

/** OpenAI strict json_schema needs additionalProperties:false + every key required. */
function toOpenAiSchema(schema: JsonSchema): JsonSchema {
  const s = JSON.parse(JSON.stringify(schema));
  if (s.type === 'object') {
    s.additionalProperties = false;
    if (s.properties && !s.required) s.required = Object.keys(s.properties);
  }
  return s;
}

async function callOpenAi(choice: ModelChoice, system: string, input: string, schema: JsonSchema, schemaName: string): Promise<string> {
  const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });
  const res = await client.responses.create({
    model: choice.model,
    instructions: system,
    input,
    text: { format: { type: 'json_schema', name: schemaName, strict: true, schema: toOpenAiSchema(schema) } },
  } as any);
  return (res as any).output_text || '';
}

async function callGemini(choice: ModelChoice, system: string, input: string, schema: JsonSchema, _schemaName: string): Promise<string> {
  const client = getGeminiClient();
  const res = await client.models.generateContent({
    model: choice.model,
    contents: [{ role: 'user', parts: [{ text: input }] }],
    config: { systemInstruction: system, temperature: 0, responseMimeType: 'application/json', responseSchema: schema },
  });
  return (res as any).text || '';
}

export async function extractStructured<T>(call: StructuredCall<T>, deps: LlmDeps = {}): Promise<StructuredResult<T>> {
  const cfg = laneConfig(call.lane);
  const oai = deps.openaiCall || callOpenAi;
  const gem = deps.geminiCall || callGemini;
  const run = (c: ModelChoice) => (c.provider === 'openai' ? oai : gem)(c, call.system, call.input, call.schema, call.schemaName);

  try {
    const raw = await run(cfg.primary);
    return { data: parseJson<T>(raw), provider: cfg.primary.provider, model: cfg.primary.model, usedFallback: false, raw };
  } catch (e) {
    console.warn(`[llm:${call.lane}] primary ${cfg.primary.model} failed, falling back:`, (e as any)?.message || e);
    try {
      const raw = await run(cfg.fallback);
      return { data: parseJson<T>(raw), provider: cfg.fallback.provider, model: cfg.fallback.model, usedFallback: true, raw };
    } catch (e2) {
      console.error(`[llm:${call.lane}] both providers failed:`, (e2 as any)?.message || e2);
      return { data: null, provider: cfg.fallback.provider, model: cfg.fallback.model, usedFallback: true, raw: '' };
    }
  }
}
```
